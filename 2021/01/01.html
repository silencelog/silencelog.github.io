      jenkins 入门

      # Centos7

      # 下载配置npm-repo

      sudo wget -O /etc/yum.repos.d/jenkins.repo https://pkg.jenkins.io/redhat-stable/jenkins.repo
sudo rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io.key

      # 安装

      sudo yum install jenkins -y
